{% comment %} Converts relative URLs in post content to absolute URLs. Usage: {% include absolute_content.html content=post.content %} Output: absolute_content variable with converted URLs {% endcomment %} {% assign images_replacement = '="' | append: site.url | append: '/images/' %} {% assign icons_replacement = '="' | append: site.url | append: '/icons/' %} {% assign blog_replacement = '="' | append: site.url | append: '/blog/' %} {% assign contact_replacement = '="' | append: site.url | append: '/contact/' %} {% assign about_replacement = '="' | append: site.url | append: '/about/' %} {% assign absolute_content = include.content | replace: '="/images/', images_replacement | replace: '="/icons/', icons_replacement | replace: '="/blog/', blog_replacement | replace: '="/contact/', contact_replacement | replace: '="/about/', about_replacement %}
